Theorem BetaRed_SN
        {B : Type}
        {F : Type}
        (ar : F → ty B)
        (C : con B)
        (A : ty B)
  : Wf (fun (t1 t2 : tm ar C A) ⇒ betaRed t1 t2).
Proof.
Admitted.
